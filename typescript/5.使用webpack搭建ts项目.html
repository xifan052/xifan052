    配置 webpack 编译 typescript

    # 配置 webpack 编译 typescript

    # 1. 初始化项目

    • 1.1 使用npm init -y初始化项目

    • 1.2 安装npm install -D webpack webpack-cli typescript ts-loader less less-loader css-loader style-loader postcss postcss-loader postcss-preset-env

    • 1.3 创建webpack.config.js,编写 webpack 配置

    const path = require('path')

module.exports = {
  entry: './src/index.ts', // 指定入口文件

  // 指定打包文件所在目录
  output: {
    path: path.resolve(__dirname, 'dist'), // 指定打包文件的目录
    filename: 'bundle.js', // 打包后的文件名称
  },

  // 指定webpack打包时要使用的模块
  module: {
    rules: [
      {
        test: /\.ts$/,
        use: 'ts-loader',
        exclude: /node_modules/,
      },
      {
        test: /\.less$/,
        use: [
          'style-loader',
          'css-loader',
          {
            loader: 'postcss-loader',
            options: {
              postcssOptions: {
                plugins: [
                  ['postcss-preset-env', { browsers: 'last 2 versions' }],
                ],
              },
            },
          },
          'less-loader',
        ],
      },
    ],
  },
}
    • 1.4 在package.json配置build打包命令

    # 2. 配置项目插件

    • 2.1 生成项目 html 文件
    // 1. 终端输入npm i -D html-webpack-plugin
// 可以为项目生成 index.html 文件并且自动引入静态资源文件
// 2. 在webpack.config.js中配置引入html-webpack-plugin
const HTMLWebpackPlugin = require('html-webpack-plugin')

plugins: [
  new HTMLWebpackPlugin({
    template: './src/index.html', // 指定自定义index.html模板文件，没有可移除template选项
  }),
],
    • 2.2 安装 webpack 开发服务器
    // 1. 终端输入npm i -D webpack-dev-server，
// 2. 在package.json的scripts中添加运行服务命令
    "start": "webpack serve --open chrome.exe"
// 3. 终端输入npm run start运行项目
    • 2.3 打包 build 时，清空原目录文件
    // 1. 终端输入npm i -D clean-webpack-plugin
// 2. 在webpack.config.js中配置引入clean-webpack-plugin
const CleanWebpackPlugin = require('clean-webpack-plugin')

plugins: [
  new CleanWebpackPlugin(),
],
    • 2.4 配置模块引入

    当直接引入其它模块的文件进行编译的时候，会提示找不到文件

      // 在webpack.config.js文件中设置引用模块
  resolve: {
    extensions: ['.ts', '.js'], // 指定ts和js文件都可以作为模块引入
  },

    # 3. 配置 babel

    ts 自带的语法编译工具满足不了低版本浏览器的代码编译

    • 3.1 安装 babel 库
    // 终端输入npm i -D @babel/core @babel/preset-env babel-loader core-js
// @babel/core babel核心库
// @babel/preset-env 可以根据配置的目标浏览器或者运行环境来自动编译对应版本代码
// babel-loader 将代码编译为浏览器可识别代码
// core-js 模拟JS运行环境，可以让老版本浏览器运行新版本代码

// 在webpack.config.js配置babel
      rules: [
      {
        test: /\.ts$/,
        use: [
          {
            // 配置babel
            loader: 'babel-loader', // 指定加载器
            options: {
              // 设置babel
              presets: [
                // 设置预定义的环境
                [
                  '@babel/preset-env', // 指定环境的插件
                  {
                    // 配置信息
                    targets: {
                      // 指定兼容的浏览器版本
                      chrome: '88',
                      ie: '11',
                    },
                    corejs: '3', // 指定corejs版本
                    useBuiltIns: 'usage', // 使用corejs的方式 usage：表示按需加载
                  },
                ],
              ],
            },
          },
          'ts-loader',
        ],
        exclude: /node_modules/,
      },
    ],
    • 3.2 移除自执行函数的箭头函数
    // 在webpack.config.js的output添加配置
    environment: {
      arrowFunction: false,
    },

    # 4. 配置端口占用自动更换端口

    // 1. 终端输入npm i -D portfinder
// 2. 在webpack.config.js中配置引入portfinder

const portfinder = require('portfinder')
module.exports = async (api) => {
  portfinder.basePort = '8080'
  const port = await portfinder.getPortPromise()
  return {
    entry: './src/index.ts', // 指定入口文件
    // 指定打包文件所在目录
    output: {
      path: path.resolve(__dirname, 'dist'), // 指定打包文件的目录
      filename: 'bundle.js', // 打包后的文件名称
      environment: {
        arrowFunction: false,
        // const:false
      },
    },
    devServer: {
      port: port || 9522,
    },
  }
}

    # 4. 最终配置

    const path = require('path')
const HTMLWebpackPlugin = require('html-webpack-plugin')
const { CleanWebpackPlugin } = require('clean-webpack-plugin')
const portfinder = require('portfinder')

module.exports = async (api) => {
  portfinder.basePort = '8080'
  const port = await portfinder.getPortPromise()
  return {
    entry: './src/index.ts', // 指定入口文件
    // 指定打包文件所在目录
    output: {
      path: path.resolve(__dirname, 'dist'), // 指定打包文件的目录
      filename: 'bundle.js', // 打包后的文件名称
      environment: {
        arrowFunction: false,
        // const:false
      },
    },
    devServer: {
      // clientLogLevel: 'warning',
      // historyApiFallback: {
      //   rewrites: [
      //     { from: /.*/, to: path.posix.join(config.dev.assetsPublicPath, 'index.html') },
      //   ],
      // },
      // hot: true,
      // contentBase: false, // since we use CopyWebpackPlugin.
      // compress: true,
      // host: HOST || config.dev.host,
      port: port || 9522,
      // open: config.dev.autoOpenBrowser,
      // overlay: config.dev.errorOverlay
      //   ? { warnings: false, errors: true }
      //   : false,
      // publicPath: config.dev.assetsPublicPath,
      // proxy: config.dev.proxyTable,
      // quiet: true, // necessary for FriendlyErrorsPlugin
      // watchOptions: {
      //   poll: config.dev.poll,
      // }
    },
    // 指定webpack打包时要使用的模块
    module: {
      rules: [
        {
          test: /\.ts$/,
          use: [
            {
              // 配置babel
              loader: 'babel-loader', // 指定加载器
              options: {
                // 设置babel
                presets: [
                  // 设置预定义的环境
                  [
                    '@babel/preset-env', // 指定环境的插件
                    {
                      // 配置信息
                      targets: {
                        // 指定兼容的浏览器版本
                        chrome: '58',
                        ie: '8',
                      },
                      corejs: '3', // 指定corejs版本
                      useBuiltIns: 'usage', // 使用corejs的方式 usage：表示按需加载
                    },
                  ],
                ],
              },
            },
            'ts-loader',
          ],
          exclude: /node_modules/,
        },
        {
          test: /\.less$/,
          use: [
            'style-loader',
            'css-loader',
            {
              loader: 'postcss-loader',
              options: {
                postcssOptions: {
                  plugins: [
                    ['postcss-preset-env', { browsers: 'last 2 versions' }],
                  ],
                },
              },
            },
            'less-loader',
          ],
        },
      ],
    },
    plugins: [
      new CleanWebpackPlugin(),
      new HTMLWebpackPlugin({
        // title: '标题',
        template: './src/index.html',
      }),
    ],
    // 设置引用模块
    resolve: {
      extensions: ['.ts', '.js'], // 指定ts和js文件都可以作为模块引入
    },
  }
}
